generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

model User {
  id          Int      @id @default(autoincrement())
  name        String
  email       String   @unique
  password    String
  roles       String[]
  PurchaseRequests PurchaseRequest[]
  ApprovalHistories ApprovalHistory[]
}

model Product {
  id              Int          @id @default(autoincrement())
  name            String
  reference       String
  characteristics String
  photo           String?
  codification    String
  categoryId      Int
  category        Category     @relation(fields: [categoryId], references: [id])
  type            ProductType
  purchasePrice   Float
  taxes           Tax[]
  unitId          Int
  unit            Unit         @relation(fields: [unitId], references: [id])
  stockItems      StockItem[]
  
  // Relation pour les produits composés
  parentId        Int?         
  parent          Product?     @relation("ProductHierarchy", fields: [parentId], references: [id])
  children        Product[]    @relation("ProductHierarchy")
  
  // Quantité par conteneur (ex: 100 stylos dans une boîte)
  quantityPerUnit Int?
}

enum ProductType {
  PIECE_RECHANGE_JETABLE
  PIECE_RECHANGE_RECURABLE
  JETABLE
  RECURABLE
}

model Unit {
  id    Int      @id @default(autoincrement())
  name  String
  tags  String[]
  products Product[]
}

model Category {
  id      Int      @id @default(autoincrement())
  name    String
  products Product[]
}

model Tax {
  id        Int      @id @default(autoincrement())
  name      String
  type      TaxType
  value     Float
  products  Product[]
  quotations QuotationItem[]
}

enum TaxType {
  PERCENTAGE
  ADDITIONAL
}

model Supplier {
  id          Int      @id @default(autoincrement())
  name        String
  phone       String
  email       String
  bankRIB     String
  address     String
  contracts   SupplierContract[]
  quotations  Quotation[]
}

model SupplierContract {
  id          Int      @id @default(autoincrement())
  supplierId  Int
  supplier    Supplier @relation(fields: [supplierId], references: [id])
  startDate   DateTime
  endDate     DateTime
  contractItems ContractItem[]
  bonsDeCommande  BonDeCommande[]
}

model ContractItem {
  id            Int      @id @default(autoincrement())
  contractId    Int
  contract      SupplierContract @relation(fields: [contractId], references: [id])
  productId     Int
  product       Product @relation(fields: [productId], references: [id])
  price         Float
  quantity      Int
  consumedQty   Int @default(0)
  taxes         Tax[]
}

model PurchaseRequest {
  id          Int      @id @default(autoincrement())
  createdById Int
  createdBy   User     @relation(fields: [createdById], references: [id])
  status      RequestStatus
  remarks     String?
  conditions  String?
  items       PurchaseRequestItem[]
  approvalFlow ApprovalFlow?
  histories   RequestHistory[]

  devis              Devis[]            
  bonsDeCommande     BonDeCommande[] 
}

enum RequestStatus {
  PENDING
  APPROVABLE
  APPROVED
  REJECTED
}

model PurchaseRequestItem {
  id          Int      @id @default(autoincrement())
  requestId   Int
  request     PurchaseRequest @relation(fields: [requestId], references: [id])
  productId   Int?
  product     Product? @relation(fields: [productId], references: [id])
  customName  String?
  customCharacteristics String?
  photo       String?
  quantity    Int
  unitId      Int
  unit        Unit @relation(fields: [unitId], references: [id])
}

model ApprovalFlow {
  id          Int      @id @default(autoincrement())
  name        String
  levels      ApprovalLevel[]
  purchaseRequests PurchaseRequest[]
}

model ApprovalLevel {
  id          Int      @id @default(autoincrement())
  flowId      Int
  flow        ApprovalFlow @relation(fields: [flowId], references: [id])
  order       Int
  users       User[]
  roles       String[]
}

model ApprovalHistory {
  id          Int      @id @default(autoincrement())
  requestId   Int
  request     PurchaseRequest @relation(fields: [requestId], references: [id])
  levelId     Int
  level       ApprovalLevel @relation(fields: [levelId], references: [id])
  approvedById Int
  approvedBy  User @relation(fields: [approvedById], references: [id])
  date        DateTime @default(now())
  status      ApprovalStatus
}

enum ApprovalStatus {
  APPROVED
  REJECTED
}

model BonDeCommande {
  id              Int       @id @default(autoincrement())
  createdAt       DateTime  @default(now())
  createdById     Int
  createdBy       User      @relation(fields: [createdById], references: [id])
  supplierId      Int
  supplier        Supplier  @relation(fields: [supplierId], references: [id])
  status          BonDeCommandeStatus
  remarks         String?
  items           BonDeCommandeItem[]
  bonDeReception  BonDeReception? @relation(fields: [id], references: [bonDeCommandeId])

// Clé étrangère vers DemandeAchat
  demandeAchatId Int
  demandeAchat   DemandeAchat @relation(fields: [demandeAchatId], references: [id])

  // Clé étrangère vers Devis
  devisId        Int
  devis          Devis       @relation(fields: [devisId], references: [id])

// Clé étrangère vers Contrat (optionnel ou obligatoire)
  contratId       Int?
  contrat         Contrat?     @relation(fields: [contratId], references: [id])
}

enum BonDeCommandeStatus {
  EN_ATTENTE
  VALIDE
  REJETE
}

model BonDeCommandeItem {
  id              Int      @id @default(autoincrement())
  bonDeCommandeId Int
  bonDeCommande   BonDeCommande @relation(fields: [bonDeCommandeId], references: [id])
  productId       Int
  product         Product @relation(fields: [productId], references: [id])
  quantity        Int
  unitPrice       Float
  taxes           Tax[]
}


model Quotation {
  id          Int      @id @default(autoincrement())
  supplierId  Int
  supplier    Supplier @relation(fields: [supplierId], references: [id])
  requestId   Int
  request     PurchaseRequest @relation(fields: [requestId], references: [id])
  items       QuotationItem[]
  notes       String?
  versions    QuotationVersion[]

// Clé étrangère vers DemandeAchat
  demandeAchatId   Int
  demandeAchat     DemandeAchat   @relation(fields: [demandeAchatId], references: [id])

  // Relation vers BonDeCommande
  bonDeCommande    BonDeCommande?
}

model QuotationItem {
  id          Int      @id @default(autoincrement())
  quotationId Int
  quotation   Quotation @relation(fields: [quotationId], references: [id])
  productId   Int?
  product     Product? @relation(fields: [productId], references: [id])
  customName  String?
  price       Float
  quantity    Int
  taxes       Tax[]
}

model QuotationVersion {
  id          Int      @id @default(autoincrement())
  quotationId Int
  quotation   Quotation @relation(fields: [quotationId], references: [id])
  version     Int
  items       QuotationItem[]
  notes       String?
  createdAt   DateTime @default(now())
}

model Stock {
  id          Int      @id @default(autoincrement())
  name        String

  managers    User[]
  stockItems  StockItem[]
}

model StockItem {
  id          Int      @id @default(autoincrement())

model RequestHistory {
  id          Int      @id @default(autoincrement())
  requestId   Int

  modifiedById Int
  modifiedBy  User @relation(fields: [modifiedById], references: [id])

}

model BonDeSortie {

model ProductPriceHistory {
  id          Int       @id @default(autoincrement())
  productId   Int
  product     Product   @relation(fields: [productId], references: [id])
  price       Float

  modifiedById Int
  modifiedBy  User      @relation(fields: [modifiedById], references: [id])
}

  CONTRACT
}
